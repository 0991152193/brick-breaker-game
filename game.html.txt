    <script>
        // 游戏核心变量
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        
        // 设置画布尺寸
        function resizeCanvas() {
            const container = document.getElementById('game-container');
            canvas.width = container.clientWidth;
            canvas.height = container.clientHeight;
        }
        
        resizeCanvas();
        window.addEventListener('resize', resizeCanvas);
        
        // 游戏状态
        let gameState = {
            isPlaying: false,
            score: 0,
            paddle: {
                x: canvas.width / 2 - 50,
                y: canvas.height - 30,
                width: 100,
                height: 15,
                speed: 8
            },
            ball: {
                x: canvas.width / 2,
                y: canvas.height / 2,
                radius: 8,
                dx: 4,
                dy: -4,
                speed: 4
            },
            bricks: [],
            brickConfig: {
                rows: 5,
                cols: 8,
                width: 50,
                height: 20,
                padding: 10,
                offsetTop: 50,
                offsetLeft: 20
            },
            doubleScore: false, // 是否开启双倍分数（看广告获得）
            lives: 3 // 生命值
        };

        // 初始化砖块
        function initBricks() {
            gameState.bricks = [];
            for (let r = 0; r < gameState.brickConfig.rows; r++) {
                gameState.bricks[r] = [];
                for (let c = 0; c < gameState.brickConfig.cols; c++) {
                    const x = c * (gameState.brickConfig.width + gameState.brickConfig.padding) + gameState.brickConfig.offsetLeft;
                    const y = r * (gameState.brickConfig.height + gameState.brickConfig.padding) + gameState.brickConfig.offsetTop;
                    gameState.bricks[r][c] = { x, y, status: 1 };
                }
            }
        }

        // 绘制球拍
        function drawPaddle() {
            ctx.fillStyle = '#4CAF50';
            ctx.fillRect(
                gameState.paddle.x,
                gameState.paddle.y,
                gameState.paddle.width,
                gameState.paddle.height
            );
            ctx.strokeStyle = '#fff';
            ctx.strokeRect(
                gameState.paddle.x,
                gameState.paddle.y,
                gameState.paddle.width,
                gameState.paddle.height
            );
        }

        // 绘制小球
        function drawBall() {
            ctx.beginPath();
            ctx.arc(gameState.ball.x, gameState.ball.y, gameState.ball.radius, 0, Math.PI * 2);
            ctx.fillStyle = '#FF9800';
            ctx.fill();
            ctx.closePath();
        }

        // 绘制砖块
        function drawBricks() {
            for (let r = 0; r < gameState.brickConfig.rows; r++) {
                for (let c = 0; c < gameState.brickConfig.cols; c++) {
                    if (gameState.bricks[r][c].status === 1) {
                        const brick = gameState.bricks[r][c];
                        ctx.fillStyle = `hsl(${r * 30}, 70%, 50%)`;
                        ctx.fillRect(brick.x, brick.y, gameState.brickConfig.width, gameState.brickConfig.height);
                        ctx.strokeStyle = '#fff';
                        ctx.strokeRect(brick.x, brick.y, gameState.brickConfig.width, gameState.brickConfig.height);
                    }
                }
            }
        }

        // 碰撞检测
        function collisionDetection() {
            // 砖块碰撞
            for (let r = 0; r < gameState.brickConfig.rows; r++) {
                for (let c = 0; c < gameState.brickConfig.cols; c++) {
                    const brick = gameState.bricks[r][c];
                    if (brick.status === 1) {
                        if (
                            gameState.ball.x > brick.x &&
                            gameState.ball.x < brick.x + gameState.brickConfig.width &&
                            gameState.ball.y > brick.y &&
                            gameState.ball.y < brick.y + gameState.brickConfig.height
                        ) {
                            gameState.ball.dy = -gameState.ball.dy;
                            brick.status = 0;
                            
                            // 分数计算（支持双倍分数）
                            const scoreAdd = gameState.doubleScore ? 20 : 10;
                            gameState.score += scoreAdd;
                            document.getElementById('score').textContent = gameState.score;
                            
                            // 检查是否通关
                            if (isGameWin()) {
                                showGameOver('恭喜通关！', true);
                            }
                        }
                    }
                }
            }

            // 墙壁碰撞
            // 左右墙壁
            if (gameState.ball.x + gameState.ball.radius > canvas.width || gameState.ball.x - gameState.ball.radius < 0) {
                gameState.ball.dx = -gameState.ball.dx;
            }
            // 上墙壁
            if (gameState.ball.y - gameState.ball.radius < 0) {
                gameState.ball.dy = -gameState.ball.dy;
            }
            // 下边界（游戏结束）
            if (gameState.ball.y + gameState.ball.radius > canvas.height) {
                gameState.lives--;
                if (gameState.lives <= 0) {
                    showGameOver('游戏结束！', false);
                } else {
                    resetBall();
                }
            }

            // 球拍碰撞
            if (
                gameState.ball.y + gameState.ball.radius > gameState.paddle.y &&
                gameState.ball.x > gameState.paddle.x &&
                gameState.ball.x < gameState.paddle.x + gameState.paddle.width
            ) {
                // 计算碰撞角度，增加游戏趣味性
                const collisionPoint = (gameState.ball.x - (gameState.paddle.x + gameState.paddle.width / 2)) / (gameState.paddle.width / 2);
                const angle = collisionPoint * Math.PI / 3;
                gameState.ball.dx = gameState.ball.speed * Math.sin(angle);
                gameState.ball.dy = -gameState.ball.speed * Math.cos(angle);
            }
        }

        // 检查是否通关
        function isGameWin() {
            let totalBricks = gameState.brickConfig.rows * gameState.brickConfig.cols;
            let brokenBricks = 0;
            
            for (let r = 0; r < gameState.brickConfig.rows; r++) {
                for (let c = 0; c < gameState.brickConfig.cols; c++) {
                    if (gameState.bricks[r][c].status === 0) {
                        brokenBricks++;
                    }
                }
            }
            
            return brokenBricks === totalBricks;
        }

        // 重置小球位置
        function resetBall() {
            gameState.ball.x = canvas.width / 2;
            gameState.ball.y = canvas.height / 2;
            gameState.ball.dx = 4 * (Math.random() > 0.5 ? 1 : -1);
            gameState.ball.dy = -4;
        }

        // 显示游戏结束界面
        function showGameOver(text, isWin) {
            gameState.isPlaying = false;
            document.getElementById('game-over-text').textContent = text;
            document.getElementById('final-score').textContent = gameState.score;
            document.getElementById('game-over-screen').style.display = 'flex';
            
            // 显示广告（游戏结束是重要的广告展示点）
            showAd('post-game');
        }

        // 广告展示函数（对接广告联盟的核心函数）
        function showAd(adType) {
            console.log(`展示${adType}类型广告`);
            // 这里替换为实际的广告调用代码
            // 示例：Google AdSense 激励视频广告
            /*
            if (typeof adsense !== 'undefined') {

                        if (adType === 'start-double-score') {
                            gameState.doubleScore = true;
                        } else if (adType === 'revive') {
                            gameState.lives = 1;
                            resetBall();
                            document.getElementById('game-over-screen').style.display = 'none';
                            gameState.isPlaying = true;
                            gameLoop();
                        }

                    }

            */
            
            // 模拟广告播放（实际开发中删除这部分）
            alert(`【广告模拟】播放${adType}广告，3秒后关闭`);
            
            // 广告播放成功的回调逻辑
            if (adType === 'start-double-score') {
                gameState.doubleScore = true;
                startGame();
            } else if (adType === 'revive') {
                gameState.lives = 1;
                resetBall();
                document.getElementById('game-over-screen').style.display = 'none';
                gameState.isPlaying = true;
                gameLoop();
            }
        }

        // 开始游戏
        function startGame() {
            // 重置游戏状态
            gameState.isPlaying = true;
            gameState.score = 0;
            gameState.lives = 3;
            document.getElementById('score').textContent = 0;
            document.getElementById('start-screen').style.display = 'none';
            document.getElementById('game-over-screen').style.display = 'none';
            
            // 初始化游戏元素
            initBricks();
            resetBall();
            
            // 开始游戏循环
            gameLoop();
        }

        // 游戏主循环
        function gameLoop() {
            if (!gameState.isPlaying) return;
            
            // 清空画布
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // 绘制游戏元素
            drawBricks();
            drawPaddle();
            drawBall();
            
            // 碰撞检测
            collisionDetection();
            
            // 更新小球位置
            gameState.ball.x += gameState.ball.dx;
            gameState.ball.y += gameState.ball.dy;
            
            // 继续循环
            requestAnimationFrame(gameLoop);
        }

        // 控制球拍移动（鼠标/触摸）
        canvas.addEventListener('mousemove', (e) => {
            const rect = canvas.getBoundingClientRect();
            const mouseX = e.clientX - rect.left;
            
            // 限制球拍在画布内
            if (mouseX > 0 && mouseX < canvas.width) {
                gameState.paddle.x = mouseX - gameState.paddle.width / 2;
            }
        });

        // 移动端触摸控制
        canvas.addEventListener('touchmove', (e) => {
            e.preventDefault();
            const rect = canvas.getBoundingClientRect();
            const touchX = e.touches[0].clientX - rect.left;
            
            if (touchX > 0 && touchX < canvas.width) {
                gameState.paddle.x = touchX - gameState.paddle.width / 2;
            }
        });

        // 按钮事件绑定
        document.getElementById('start-btn').addEventListener('click', () => {
            gameState.doubleScore = false;
            startGame();
        });

        document.getElementById('start-with-ad-btn').addEventListener('click', () => {
            showAd('start-double-score');
        });

        document.getElementById('restart-btn').addEventListener('click', () => {
            gameState.doubleScore = false;
            startGame();
        });

        document.getElementById('revive-with-ad-btn').addEventListener('click', () => {
            showAd('revive');
        });

        // 初始化
        initBricks();
    </script>
